m = folium.Map(location=[38,135], zoom_start=5)
folium.Choropleth(
    geo_data=geo,
    key_on='feature.properties.name',
    data=df1,
    columns=['prefecture', 'elem_male'],
    fill_color='BuPu',
    legend_name='elem_male'
).add_to(m)
m.save('elem_male.html')
display(m)
